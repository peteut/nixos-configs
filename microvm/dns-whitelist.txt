# Title:       microvm DNS whitelist
# Description: Domains that microvm guests are allowed to resolve.
#              Subdomains are automatically included by dnsmasq.
# Format:      One domain per line.  Lines starting with # are comments.
#              Blank lines are ignored.  Comments must start at column 0.
#
# Usage in host config:
#
#   modules.microvm.dnsFilter = {
#     enable = true;
#     whitelistFile = ./microvm/dns-whitelist.txt;
#   };
